def info(site)
    i = 0
    array = []
    rss = HTTParty.get(site, {headers: {"User-Agent" => "Httparty"}}).body
    feed = RSS::Parser.parse(rss)
    feed.items.each do |item|
        if i.to_i <=  2
            array += [[item.title, item.link]]
            i += 1
        end
    end
array
end
